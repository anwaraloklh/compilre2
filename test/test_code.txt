  </div>

  <div id="addPage" style="display:none;">
    <h2>Add Product</h2>
    <form onsubmit="addProduct(event)">
      <input id="newName" placeholder="Name" required>
      <input id="newImage" placeholder="Image URL" required>
      <textarea id="newDetails" placeholder="Details" rows="3" required></textarea>
      <button type="submit">Add</button>
    </form>
  </div>

  <div id="detailsPage" style="display:none;">
    <h2>Product Details</h2>
    <img id="detailImage">
    <h2 id="detailName"></h2>
    <p id="detailText"></p>
    <button onclick="goTo('list')">Back</button>
    <button id="deleteBtn" style="background:darkred;">Delete</button>
  </div>

  <script>
    let products = [
      { name: 'Product 1', source: 'assets/Image/1.jpg', details: 'Details of Product 1.' },
      { name: 'Product 2', source: 'assets/Image/2.jpg', details: 'Details of Product 2.' },
      { name: 'Product 3', source: 'assets/Image/3.jpg', details: 'Details of Product 3.' }
    ];

    let selectedIndex = null;

    function renderProducts() {
      const container = document.getElementById('productsContainer');
      container.innerHTML = '';
      products.forEach((p, i) => {
        const card = document.createElement('div');
        card.className = 'product-card';
        card.innerHTML = `
          <p>${p.name}</p>
          <img src="${p.source}" alt="${p.name}">
        `;
        card.onclick = () => viewDetails(i);
        container.appendChild(card);
      });
    }

    function goTo(page) {
      document.getElementById('listPage').style.display = (page === 'list') ? 'block' : 'none';
      document.getElementById('addPage').style.display = (page === 'add') ? 'block' : 'none';
      document.getElementById('detailsPage').style.display = (page === 'details') ? 'block' : 'none';
      if (page === 'list') renderProducts();
    }

    function addProduct(event) {
      event.preventDefault();
      const name = document.getElementById('newName').value;
      const image = document.getElementById('newImage').value;
      const details = document.getElementById('newDetails').value;
      products.push({ name, source: image, details });
      document.getElementById('newName').value = '';
      document.getElementById('newImage').value = '';
      document.getElementById('newDetails').value = '';
      goTo('list');
    }

    function viewDetails(index) {
      selectedIndex = index;
      const p = products[index];
      document.getElementById('detailImage').src = p.source;
      document.getElementById('detailName').innerText = p.name;
      document.getElementById('detailText').innerText = p.details;
      document.getElementById('deleteBtn').onclick = () => deleteProduct(index);
      goTo('details');
    }

    function deleteProduct(index) {
      products.splice(index, 1);
      goTo('list');
    }


    renderProducts();
  </script>
